# -*- coding: utf-8 -*-
'''
上传文件夹到七牛
'''
from qiniu import Auth, put_file
import os
import traceback
#export QNACCKEY=${QNAKIFI}
#export QNSECKEY=${QNSKIFI}
#export QNBUCKET=${QNBKIFI}
AK = os.environ['QNACCKEY']
SK = os.environ['QNSECKEY']

# local absoule path 要上传的文件夹绝对路径
dir = '/mnt/c/Users/code0/Documents/Axure/HTML/Promo_reco'
# qiniu release path,it will replace default release path 
rlspath='lolo' 
# local replace path 需要替换的前缀
rplpath='c/Users/code0/Documents/Axure/HTML/Promo_reco'
# qiniu bucket name
bucket_name = os.environ['QNBUCKET']

q = Auth(AK, SK)
token = q.upload_token(bucket_name)

def updir(dirpath):
    if os.path.isdir(dirpath):#文件夹
        sublist = os.listdir(dirpath)
        for sub in sublist:
            updir(dirpath+'/'+sub)
    else:#文件
        fpath, fname = os.path.split(dirpath)
        patharr = fpath.split('/')
        try:
            key = getKey(dirpath)
            ret =0
            #print('key-->'+key)
            ret, info = put_file(token, key, dirpath)
            print(ret)
        except:
            traceback.print_exc()

def getKey(file):
    key = ''
    fpath, fname = os.path.split(file)
    patharr = fpath.split('/')
    if len(patharr) >= 2:
    #    key = '' + fname
        #key = '/'.join(patharr[(len(patharr)-3):]) + '/' + fname
        key = '/'.join(patharr[2:]) + '/' + fname
        key = key.replace(rplpath,rlspath)
    else:
        key = fname
    print('key-g-->'+key)    
    return key

if __name__ == '__main__':
    updir(dir)
